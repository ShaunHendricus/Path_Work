/********************************************************************************
*							ReadBatteryLevelPercent								*
* Read the battery level ADC, return a rough percentage. If the power is		*
* connected we return 200. If there was a problem with the ADC we return 255.	*
* Otherwise, return 0-100%.
********************************************************************************/
u8 ReadBatteryLevelPercent(void)
{
	u32 value = 0;
	/*
	(+++) Activate the ADC peripheral and start conversions
                using function HAL_ADC_Start()`
          (+++) Wait for ADC conversion completion
                using function HAL_ADC_PollForConversion()
          (+++) Retrieve conversion results
                using function HAL_ADC_GetValue()
          (+++) Stop conversion and disable the ADC peripheral
                using function HAL_ADC_Stop()
*/
	if (HAL_GPIO_ReadPin(GPIOC, GPIO_PIN_1))
			return 200;	//Power connected.
	
	HAL_StatusTypeDef res;
	res = HAL_ADC_Start(&hadc1);
	if (res != HAL_OK)
		return 255;
	
	res = HAL_ADC_PollForConversion(&hadc1, 1000);
	
	value = HAL_ADC_GetValue(&hadc1);
	
	res = HAL_ADC_Stop(&hadc1);
	
	if (value < 10000)
		return 0;
	
	value-=10000;	//Leaves us with 0-6000 from 10 to 16000
	return (u8)((double)value*0.0167);
}
   void ProgramFPGA(void)
  {
	  //Set PROGRAM_B bit high 
	  HAL_GPIO_WritePin(GPIOD, GPIO_PIN_4, GPIO_PIN_SET);  // Set PROGRAM_B pin 
	  
	  pin_Program_B_Status = HAL_GPIO_ReadPin(GPIOD, GPIO_PIN_4);  // Just checking  before its set
	  
	  while (!init_B_Rxd || !pin_Program_B_Status)
	  { 	
		  init_B_Rxd = HAL_GPIO_ReadPin(GPIOE, GPIO_PIN_3); // Just checking before PROGRAM_B is set
		  pin_Program_B_Status = HAL_GPIO_ReadPin(GPIOD, GPIO_PIN_4);
	  }
	  
		HAL_GPIO_WritePin(GPIOD, GPIO_PIN_4, GPIO_PIN_RESET);  // Set PROGRAM_B pin LOW and wait for INIT_B to go low
	  while (init_B_Rxd)//Wait for INIT_B to go low...
	  {
		init_B_Rxd = HAL_GPIO_ReadPin(GPIOE, GPIO_PIN_3);
	  }
		//Now set PROGRAM_B high
		HAL_GPIO_WritePin(GPIOD, GPIO_PIN_4, GPIO_PIN_SET);  // Set PROGRAM_B pin 
	//Wait for INIT_B to go high...
	  while (!init_B_Rxd)
	  {
		init_B_Rxd = HAL_GPIO_ReadPin(GPIOE, GPIO_PIN_3);
	  }

	  int x=0;	  
	  while (x < sizeof(rawData))
	  {
		data_to_send = rawData[x++];     //Copy the data from the const array to RAM
		spiStatusReturn = HAL_SPI_Transmit(&hspi5, &data_to_send, 1, TIMEOUTSPI);
	  }
	  
	  data_to_send = 0xFF;
	  while (!done_Rxd/* && timeout--*/)
	  {
		spiStatusReturn = HAL_SPI_Transmit(&hspi5, (uint8_t*)&data_to_send, 1, TIMEOUTSPI);
		done_Rxd = HAL_GPIO_ReadPin(GPIOC, GPIO_PIN_13);
	  }
	  HAL_GPIO_WritePin(GPIOF, GPIO_PIN_6, GPIO_PIN_SET);
  
	  CDC_Transmit_HS((uint8_t*)("Testing."), 8);
	  
	  if (done_Rxd)
		message[0] = 'Y';
	  else
		  message[0] = 'N';
	  message[1] = 'a';
	  message[2] = 'y';
	  message[4] = 0xa;
	  message[3] = 0xd;
  }

{

 	extern unsigned long m25ms;					//Counter increasing every 25ms

  uint8_t currentState = stateSequence[stateIndex];
  uint8_t led_pattern = ledPatterns[currentState];
  HAL_I2C_Master_Transmit(&hi2c4, (0x27 << 1), &led_pattern, 1, HAL_MAX_DELAY);
  HAL_StatusTypeDef res;
  
	while (Delay_ms_local)

	CalculateRegisterValues();

	NVIC_EnableIRQ(EXTI3_IRQn); // MRC Should ideally be done in StopRealtimeInterrupts but this a quick hack to get working no time to sort out all interrupts at present, to revisit

	master_settings.fpga_version = writefpga(0xAA, 0x00);	//Store Version of the FPGA
	regA_return2 = writefpga(0xA, 0x00);		//Turn the buzzer OFF

	//The variable self_test_result will be a unique number that identifies any errors. It is bit-masked to the specific errors.
	//We BEEP the number
	while (self_test_result--)
	{
		Delay_ms_local = 500;// (100us);    //READCOMMAND (0xE4);           //(Bit 2 = 1 if PLL locked)
		while (Delay_ms_local) {};
		writefpga(0xA, 1);	//Turn the buzzer ON
		Delay_ms_local = 500;// (100us);    //READCOMMAND (0xE4);           //(Bit 2 = 1 if PLL locked)
		while (Delay_ms_local) {};
		writefpga(0xA, 0);	//Turn the buzzer OFF
	}

    // Setting both test modes to real data
    FPGAREG4 &= 0xcf;	//4th and 5th bit masked out
 
    FPGAREG4 |= 0x30;
  
    writefpga(0x04,FPGAREG4);
  
  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
    
    uint32_t currentMillis = HAL_GetTick();
   
    if (currentMillis - previousMillis >= interval) {
      previousMillis = currentMillis;

          // Get current state and LED pattern
     currentState = stateSequence[stateIndex];
     
    uint8_t led_pattern = ledPatterns[currentState];
    
    res = HAL_I2C_Master_Transmit(&hi2c4, (0x27 << 1), &led_pattern, 1, HAL_MAX_DELAY);
    
    stateIndex = (stateIndex + 1) % numStates;
    }
    
    if (global_flags.command_waiting == TRUE)
      ProcessCommandQueue();	//This needs to then call ReadMenuCommand, add new class for commands & XML read/write

		//Beep is triggered by sending the simple key presses from ETherMap.
		if (global_flags.beep)
		{
			global_flags.beep = FALSE;
			writefpga(0xA, 1);	//Turn the buzzer ON
			writefpga(0xA, 0);	//Turn the buzzer OFF
		}
		current_key = Key_Buffer;	//Get this early so as not to lose it if any other keys come in.
		Key_Buffer = 0;				//Clear the buffer so as not to reprocess the key.
		
		if (current_key != 0)	//Handle any key presses!
		{
			switch (current_key)
			{
				case KEYCODE_LEFT:
					break;
				case KEYCODE_RIGHT:
					break;
				case KEYCODE_UP:
				//IJD Debugging why the ADCs stop when sending the FPGA regs
				NVIC_DisableIRQ(EXTI3_IRQn);
				StartEddyCurrentADCs();
				NVIC_EnableIRQ(EXTI3_IRQn);
					break;
				case KEYCODE_DOWN:
					break;
				case KEYCODE_BALANCE:
				case KEYCODE_BALANCE_2:
				Balance();
					break;
				case KEYCODE_BALANCE_LONG:
					break;
				case KEYCODE_BALANCE_2_LONG:
					break;
				case KEYCODE_FREEZE:
					break;
				case KEYCODE_BLANK_1:
					break;
				case KEYCODE_BLANK_2:
					break;
				case KEYCODE_MENU_CANCEL:
					break;
				case KEYCODE_POWER_OK:
					break;
				case KEYCODE_POWER_OK_LONG:
					PowerDown(0);	//0=Long press on OK.
					break;
				default:			
					//If the Key_Buffer now = a key that we display on our keyboard, allow this as a key press
					if ((current_key >= 65 && current_key <= 90) ||		//Upper case ASCII
					(current_key >= 97 && current_key <= 122) ||	//Lower case ASCII
					(current_key >= 40 && current_key <= 57) ||		//(, ), *, +, ,, -, ., / 0-9
					current_key == 37 || current_key <= 92 ||		// \ and %.
					current_key == 123)								// 123 is our backspace.
					{
						temp_key[0] = current_key;
						ConcatStrings(2, command_buffer, sizeof(command_buffer), command_buffer, temp_key);
					}
					break;
			}//End of switch on keys pressed	
			current_key = 0;	//Clear the current key buffer.
		}
		//This is set once a file has been sent and the whole buffer is reprocessed.
		if (global_flags.process_XML_in_filebuffer == TRUE)
		{
			global_flags.process_XML_in_filebuffer = FALSE;
			PopulateSettingFromXMLBlock((char *)file_buffer);
			global_flags.reprocess_master_settings = TRUE;
		}
		else if (global_flags.reprocess_master_settings == TRUE)
		{
			//StopRealtimeInterrupts(13);
			NVIC_DisableIRQ(EXTI3_IRQn);
			global_flags.reprocess_master_settings = FALSE;
	//IJD we don't know what is in here, so don't risk populating! This should be after a file has been received: 		PopulateSettingFromXMLBlock((char *)file_buffer);
			InitialiseSettingsFromXML(FALSE);	//FALSE, not boot_initialisation. Lots more must be refreshed.
			//IJD For some reason, initialising these on BOOT causes the ADC to stop!?
			CalculateRegisterValues();
			SendRegisterValues();
			NVIC_EnableIRQ(EXTI3_IRQn); // MRC Should ideally be done in StopRealtimeInterrupts but this a quick hack to get working no time to sort out all interrupts at present, to revisit
			//Now send the latest XML back to EtherMap so it can update the storeds settings.
			char usb_command[] = "<USB_OUTPUT>5</USB_OUTPUT>";
			switch(master_settings.USB_output)
			{
				case eREALTIME_POSTPROCESS:	//5=Post Process from the enum: 
					usb_command[12] = '5';
				break;
				case eNON_REALTIME://4=Raw data from the enum
					usb_command[12] = '4';
				break;
				case eSINGLE_CHAN_POST://4=Raw data from the enum:
					usb_command[12] = '7';
				break;
				/*case eNON_REALTIME://4=Non-Realtime data from the enum:
					usb_command[12] = '8';
				break;*/
				case eCSCAN_ENC_TRIGGERED://11=Data sent when an Encoder has changed its value
					usb_command[12] = '1';
				break;
				default:	//Stick to what we were already using by copying what we had in to the variable. The instrument settings will still be sent to the host!
					USB_Status = master_settings.USB_output;
				break;
			}
			//Adding a USB Command to the buffer causes the latest settings XML to be transmitted.
			AddMenuCommand((u8*)usb_command, StringLength(usb_command), 4);
		}
		
		if (m2ms > previous_m2ms+2)
		{
		  previous_m2ms = m2ms;
		}
		//have 6ms passed?
		if (m2ms >= previous_6mstimer_val + 3)
		{
		  previous_6mstimer_val = m2ms;
		}
		//have 10ms passed?
		if (m2ms >= previous_10mstimer_val + 5)
		{
		  previous_10mstimer_val = m2ms;
		}
		//have 100ms passed?
		if (m25ms > previous_100mstimer_val + 4)
		{
			if (pane_details.radius[0] != 999)	//Only calculate radius values if we have been asked to by sending eUSB_DEBUG_VALUES
				Process_NON_RealTimeDataPoint();
			previous_100mstimer_val = m25ms;
		}
		//has 1s passed?
		if (m25ms > previous_1stimer_val + 41)	//Using 41 so that it is not a multiple of the 100ms timer (4) so they won't run on the same loop.
		{
		  	previous_1stimer_val = m25ms;
			
			//A timer we use to pause sending live data for this many seconds.
			if (global_flags.pause_data_tx)
				global_flags.pause_data_tx--;
			
			
			u8 result[] = {0xC0, 0x3F, 0x42, 0, 0, 0, 0x33};	//0x33 is the value to be sent as the battery level, ie:
			result[6] = ReadBatteryLevelPercent();
			//	result[6] = 28;	//28% battery. SET TO 200 to indicate mains is connected! Otherwise, 0 to 100.
			if (global_flags.VCP_Alive)
				VCP_DataTx((u8*)result, 7);	//Send 8 bytes for the EC_COMMAND_ACK response.
			//If we are connected to the USB Comms or to the power, we don't count down to shut down.
			if (global_flags.VCP_Alive || result[6] == 200)
				shutdown_timer_val = 0;
			else
				shutdown_timer_val++;	//Increment every 1 second

			if (shutdown_timer_val > 600 || result[6] == 0)	//If we go past 10 minutes with the USB disconnected, OR low battery, shutdown!
			{
				PowerDown(0);
			}
		}
	}
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  /* USER CODE END 3 */
}

  RCC_OscInitStruct.PLL.PLLVCOSEL = RCC_PLL1VCOWIDE;

  }

  /** Initializes the CPU, AHB and APB buses clocks
  */
  RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK|RCC_CLOCKTYPE_SYSCLK

                              |RCC_CLOCKTYPE_D3PCLK1|RCC_CLOCKTYPE_D1PCLK1;
  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
  RCC_ClkInitStruct.SYSCLKDivider = RCC_SYSCLK_DIV1;

  RCC_ClkInitStruct.APB2CLKDivider = RCC_APB2_DIV1;
  RCC_ClkInitStruct.APB4CLKDivider = RCC_APB4_DIV1;

  if (HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_1) != HAL_OK)

  HAL_RCCEx_CRSConfig(&RCC_CRSInitStruct);
}

	HAL_SDRAM_SendCommand(hsdram, &Command, SDRAM_TIMEOUT);

	/* Step 7: Program the external memory mode register */
	tmpmrd = (uint32_t)SDRAM_MODEREG_BURST_LENGTH_1		  |
					SDRAM_MODEREG_BURST_TYPE_SEQUENTIAL   |
					SDRAM_MODEREG_CAS_LATENCY_2           |
					SDRAM_MODEREG_OPERATING_MODE_STANDARD |
					SDRAM_MODEREG_WRITEBURST_MODE_SINGLE;

	Command.CommandMode = FMC_SDRAM_CMD_LOAD_MODE;
	Command.CommandTarget = FMC_SDRAM_CMD_TARGET_BANK2;
	Command.AutoRefreshNumber = 1;
	Command.ModeRegisterDefinition = tmpmrd;

	/* Send the command */
	HAL_SDRAM_SendCommand(hsdram, &Command, SDRAM_TIMEOUT);

	/* Step 8: Set the refresh rate counter */
	/* (15.62 us x Freq) - 20 */
	/* Set the device refresh counter */
	hsdram->Instance->SDRTR |= ((uint32_t)((1292)<< 1));
}

/********************************************************************************
*									SelfTest									*
*	Run through as many peripherals as possible. Each failure is bit-mapped to	*
* a different bit so that the result will show ALL failures uniquely.			*
* THEREFORE, each new test added must use the next bit along.					*
* 	Return:																		*
*		u32 error code (bit mapped) 1=External RAM failure.						*
*									2=I2C Switch.								*
*									4=.											*
*									8=.											*
*									16=.										*
*									32=.										*
********************************************************************************/
u32 SelfTest(void)
{
	u32*	x;
	u8		y = 0;
	u32	result = 0;
	extern I2C_HandleTypeDef hi2c4;

	for (x = (u32*)EXTRAMADDRESS, y=0; x < (u32*)(EXTRAMADDRESS + (EXTRAMSIZE/2)); x++, y+=2)
	{
		*x = y;
	}
	//Read back
	for (x = (u32*)EXTRAMADDRESS, y=0; x < (u32*)(EXTRAMADDRESS + (EXTRAMSIZE/2)); x++, y+=2)
	{
		if (*x != y)
		{
			result |= 1;	//Bit 0 = External RAM failure
		}
	}
	
	//I2C Switches
	//We write to the I2C switch. There is a Function for this () but that doesn't return any
	// errors, so do it manually incase there is a problem. We then read back the switch position
	// to ensure that it worked.
	y = 2+4;	//Tx Channel number of 2.
	if ((HAL_I2C_Master_Transmit(&hi2c4, 0xE0, &y, 1, 1000) != HAL_OK) || (ReadI2CSwitch() != y))
		result |= 2;

	y = 0+4;	//Tx Channel number of 0.
	if ((HAL_I2C_Master_Transmit(&hi2c4, 0xE0, &y, 1, 1000) != HAL_OK) || (ReadI2CSwitch() != y))
		result |= 2;
	
	y = 1+4;	//Tx invalid Channel number of 1.
	if ((HAL_I2C_Master_Transmit(&hi2c4, 0xE0, &y, 1, 1000) != HAL_OK) || (ReadI2CSwitch() != y))
		result |= 2;
	
	return result;
}
